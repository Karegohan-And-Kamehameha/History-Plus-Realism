mend_schism_religion_trigger = {
	is_heretic = no
	OR = {
		AND = {
			OR = {
				religion = catholic
				religion = fraticelli
				religion = cathar
			}
			event_target:mender_religion = {
				OR = {
					religion = orthodox
					religion = monothelite
					religion = iconoclast
				}
			}
		}
		AND = {
			OR = {
				religion = orthodox
				religion = monothelite
				religion = iconoclast
			}
			event_target:mender_religion = {
				OR = {
					religion = catholic
					religion = fraticelli
					religion = cathar
				}
			}
		}
	}
}

				}

		}
