	convert_to_capital_religion = {
		potential = {
			is_playable = yes
			controls_religion = no
			is_nomadic = no
			mercenary = no
			capital_scope = {
				NOT = { religion = ROOT }
				religion_group = ROOT
				NOT = {
					ROOT = {
						liege = {
							religion = PREVPREV
						}
					}
				}
			}
			NOT = { has_character_flag = converted_religion }
			NOT = {
				AND = {
					has_dlc = "Charlemagne"
					has_landed_title = e_byzantium  # May instead renounce iconoclasm
					has_landed_title = c_byzantion
					religion = iconoclast
					capital_scope = { religion = orthodox }
				}
			}
		}
		allow = {
			NOT = { trait = zealous }
		}
		effect = {
			piety = -100
			if = {
				limit = { higher_tier_than = BARON }
				religion_authority = {
					modifier = ruler_converted_from
				}
			}
			capital_scope = {
				reverse_religion = ROOT
			}
			if = {
				limit = { higher_tier_than = BARON }
				hidden_tooltip = {
					religion_authority = {
						modifier = ruler_converted_to
					}
				}
			}
			hidden_tooltip = {
				any_courtier = {
					limit = {
						is_adult = no
						is_ruler = no
						dynasty = ROOT
					}
					religion = ROOT
				}
			}
			set_character_flag = converted_religion
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 0
		}
	}
