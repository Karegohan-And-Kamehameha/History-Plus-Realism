# Byzantine Empire is flourishing
narrative_event = {
	id = Plus.1300
	title = "EVTNAME_Plus_1300"
	desc = "EVTDESC_Plus_1300"
	picture = "GFX_evt_byzantine_throne"
	
	is_triggered_only = yes
	hide_from = yes
	
	trigger = {
		has_landed_title = e_byzantium
		has_global_flag = byz_empire_flourishes 
	}
	
	immediate = {
		if = {
			limit = { is_ironman = no }
			e_byzantium = { set_variable = { which = "imperial_decadence" value = 10 } }
		}
	}
	
	option = {
		name = "EVTOPTA_Plus_1300"
		clr_global_flag = byz_empire_flourishes
		if = {
			limit = { NOT = { trait = strong_basileus } }
			add_trait = weak_basileus
		}
		hidden_tooltip = {
			any_vassal = {
				limit = {
					tier = DUKE
					is_feudal = yes
					religion = ROOT
					dynasty = ROOT
					primary_title = { de_jure_liege_or_above = e_byzantium }
					NOT = { has_claim = e_byzantium }
				}
				add_weak_claim = e_byzantium
			}
		}
		# change the decadence modifier
		hidden_tooltip = { character_event = { id = Plus.1852 } }
	}
}

# New Byzantine emperor ascends to the throne
character_event = {
	id = Plus.1320
	desc = "EVTDESC_Plus_1320"
	picture = "GFX_evt_byzantine_throne"
	
	only_playable = yes
	is_triggered_only = yes

	trigger = {
		FROM = {
			title = e_byzantium
			is_primary_holder_title = yes
		}
		OR = {
			religion = orthodox
			religion = bogomilist
			religion = monothelite
			religion = iconoclast
			religion = hellenic_pagan
		}
		NOR = {
			has_character_flag = was_strong_basileus_byz 
			has_character_flag = was_weak_basileus_byz
		}	
	}
	
	immediate = {
		FROMFROM = {
			if = {
				limit = {
					is_alive = yes
					OR = {
						trait = strong_basileus
						trait = weak_basileus	
					}
				}
				if = {
					limit = { trait = strong_basileus }
					set_character_flag = was_strong_basileus_byz 
					remove_trait = strong_basileus 
				}
				if = {
					limit = { trait = weak_basileus }
					set_character_flag = was_weak_basileus_byz  
					remove_trait = weak_basileus 
				}
			}
		}
	}

	option = {
		name = "EVTOPTA_Plus_1320" #OK, then
		e_byzantium = {
			if = {
				limit = { has_law = crown_authority_4 }
				ROOT = { set_character_flag = byzantium_had_centr_four }
			}       
			if = {
				limit = { has_law = crown_authority_3 }
				ROOT = { set_character_flag = byzantium_had_centr_three }
			}                
			if = {
				limit = { has_law = crown_authority_0 }
				ROOT = { set_character_flag = byzantium_had_centr_zero }
			} 
			ROOT = { set_character_flag = centralize_without_penalties }
			if = {
				limit = {
					NOR = {
						has_law = crown_authority_1
						has_law = crown_authority_0
					}
				}
				add_law = crown_authority_1
			}
		}
		clr_character_flag = centralize_without_penaltie
	}
}
