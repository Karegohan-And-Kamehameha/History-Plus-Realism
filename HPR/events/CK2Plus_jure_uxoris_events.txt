# on_death event for usurping husband - no sons by usurped wife
character_event = {
	id = FemRule.300
	desc = EVTDESC_FemRule_300
	picture = GFX_evt_death
	
	is_triggered_only = yes
	
	trigger = {
		has_character_flag = usurped_my_wife
		any_spouse = {
			has_character_flag = usurped_by_husband
			NOT = {
				any_child = {
					father = { character = ROOT }
					NOT = { trait = bastard }
					is_female = no
					is_alive = yes
				}
			}
			any_claim = {
				any_claimant = {
					character = PREVPREV
					has_strong_claim = PREV
				}
				holder_scope = { character = ROOT }
				OR = {
					has_law = succ_gavelkind
					has_law = succ_seniority
					has_law = succ_primogeniture
					has_law = succ_ultimogeniture
				}
			}
		}
	}
	
	immediate = {
		clr_character_flag = usurped_my_wife
		hidden_tooltip = {
			any_vassal = {
				limit = {
					NOT = { is_primary_heir = ROOT }
					NOT = { any_spouse = { character = ROOT } }
				}
				character_event = { id = FemRule.302 }
			}
			current_heir = {
				character_event = { id = FemRule.302 }
			}
		}
		any_spouse = {
			limit = { has_character_flag = usurped_by_husband }
			clr_character_flag = usurped_by_husband
			hidden_tooltip = {
				any_dynasty_member = {
					limit = {
						NOT = { is_primary_heir = ROOT }
						NOT = { vassal_of = ROOT }
					}
					character_event = { id = FemRule.302 }
				}
			}
			if = {
				limit = { is_ruler = no }
				move_character = ROOT
			}
			any_claim = {
				limit = {
					any_claimant = {
						character = PREVPREV
						has_strong_claim = PREV
					}
					holder_scope = { character = ROOT }
					OR = {
						has_law = succ_gavelkind
						has_law = succ_seniority
						has_law = succ_primogeniture
						has_law = succ_ultimogeniture
					}
				}
				gain_title_plus_barony_if_unlanded = PREV
				add_weak_pressed_claim = ROOT
				hidden_tooltip = {
					PREV = { character_event = { id = FemRule.301 } }
				}
			}
		}
	}
	
	option = {
		name = EVTOPTA_FemRule_300
	}
}
