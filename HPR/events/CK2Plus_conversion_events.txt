	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion
	
	only_playable = yes

	capable_only = yes
	prisoner = no
	
	trigger = {

				}
				trait = sympathy_zoroastrianism
			}
			AND = {
				culture_group = israelite

			}
			AND = {
				NOR = {
					culture = persian
					culture = kurdish
					culture = baloch
					culture_group = israelite
				}
				trait = sympathy_pagans
			}
		}
		OR = {
			top_liege = {
				ai = yes
				religion = ROOT
				is_adult = yes
				prisoner = no
				NOT = { trait = incapable }
				NOT = { character = ROOT }
				capital_holding = { holding_type = temple }
			}
			top_liege = {
				any_realm_character = {
					ai = yes
					religion = ROOT
					is_ruler = yes

		}
		# then top liege
		if = {
			limit = {
				top_liege = {
					ai = yes
					religion = ROOT
					is_adult = yes
					prisoner = no
					NOT = { trait = incapable }
					NOT = { character = ROOT }
					capital_holding = { holding_type = temple }
				}
			}
			top_liege = { save_event_target_as = target_concerned_priest }
			break = yes
		}
		# then any realm priest of count rank or higher
		if = {
			limit = {
				top_liege = {
					any_realm_character = {
						ai = yes
						religion = ROOT
						is_ruler = yes
						higher_tier_than = BARON
