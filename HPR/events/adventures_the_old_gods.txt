# Targeted ruler receives word of the adventurer
character_event = {
	id = TOG.1201
	desc = EVTDESC_TOG_1201
	picture = GFX_evt_crusaders
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTA_TOG_1201"
		opinion = {
			who = FROM
			modifier = opinion_targeted_by_adventurer
			months = 36
		}
		hidden_tooltip = {
			# Adventurer first asks their liege for support if independent and in a different realm
			if = {
				limit = {
					NOT = { is_vassal_or_below = FROM }
					FROM = {
						liege = {
							independent = yes
							NOT = { same_realm = ROOT }
						}
					}
				}
				FROM = {
					set_character_flag = ask_from_liege_first
					character_event = { id = TOG.1231 }
				}
				break = yes
			}
			# Adventurer moves to another court to begin campaigning if liege is not independent
			if = {
				limit = {
					NOT = { is_vassal_or_below = FROM }
					FROM = {
						liege = {
							independent = no
							NOT = { same_realm = ROOT }
						}
					}
				}
				FROM = {
					character_event = { id = TOG.1231 days = 10 random = 10 }
					add_character_modifier = { name = do_not_disturb duration = 20 hidden = yes }
				}
				break = yes
			}
			# Adventurer immediately moves to another court if in the targeted ruler's realm
			if = {
				limit = {
					OR = {
						is_vassal_or_below = FROM
						FROM = { liege = { same_realm = ROOT } }
					}
				}
				FROM = { character_event = { id = TOG.1231 } }
				break = yes
			}
		}
	}
}
