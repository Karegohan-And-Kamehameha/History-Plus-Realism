	is_triggered_only = yes

		name = EVTOPTA_Plus_1200
		custom_tooltip = { text = send_invitations }
		hidden_tooltip = {
			top_liege = {
				if = {
					limit = { NOT = { character = ROOT } }
					letter_event = { id = Plus.1201 }
				}
				any_realm_character = {
					limit = {
						OR = {
							is_vassal_or_below = ROOT
							is_liege_or_above = ROOT
							higher_tier_than = BARON
							is_close_relative = ROOT
							is_allied_with = ROOT
							any_spouse = {
								OR = {
									is_liege_or_above = ROOT
									higher_tier_than = BARON
									is_close_relative = ROOT
									is_allied_with = ROOT
								}
							}
						}
						is_adult = yes
						prisoner = no
						NOR = {
							trait = incapable
							trait = excommunicated
							character = PREV
							character = ROOT
						}
					}
					letter_event = { id = Plus.1201 }
				}
			}
			any_independent_ruler = {
				limit = { NOT = { same_realm = ROOT } }
				if = {
					limit = {
						religion = ROOT
						is_adult = yes
						prisoner = no
						is_within_diplo_range = ROOT
						OR = {
							is_allied_with = ROOT
							any_realm_province = {
								ROOT = {
									any_realm_province = {
										NOT = { distance = { where = PREVPREV value = 125 } }
									}
								}
							}
						}
						NOR = {
							trait = incapable
							trait = excommunicated
							reverse_has_truce = ROOT
						}						
					}
					letter_event = { id = Plus.1201 }
				}
				any_realm_character = {
					limit = {
						OR = {
							is_allied_with = ROOT
							spouse = { is_allied_with = ROOT }
							is_close_relative = ROOT
							spouse = { is_close_relative = ROOT }
						}
						independent = no
						is_adult = yes
						prisoner = no
						is_within_diplo_range = ROOT
						NOR = {
							trait = incapable
							trait = excommunicated

		OR = {
			character = FROM

		}
	}
	
	option = {
		name = EVTOPTA_Plus_1215
		trigger = { character = FROM }

			limit = {
				OR = {

			}
			add_trait = crowned

			limit = {

					primary_title = { change_variable = { which = "imperial_decadence" value = -1 } }

				}
			}
		}
		hidden_tooltip = {
			character_event = { id = Plus.1216 days = 1 } # the feast begins
		}
		if = {
			limit = { ai = no }
			chronicle = {
				entry = CHRONICLE_NORMAL_CORONATION

				picture = GFX_evt_coronation
			}
		}
	}

		trigger = {
			NOT = { character = FROM }
			has_opinion_modifier = { who = FROM modifier = attending_coronation }
		}

			limit = { ai = no }
